tree("Attack")
	//When the enemy is visible, attack it,
	//Otherwise forget about it.
	fallback

		//Attack the enemy if visible.
		while
			canAttack
			sequence
				SetDestination_Player
				Stop
				Wait(0.5)
				AimAt_Target
				Wait(0.1)
				Attack
		sequence
			StopAttacking
			Fail

		
tree("Patrol")
	//While no enemy is spotted,
	//follow the assigned waypoints
	fallback
		while
			not canAttack
			repeat
				sequence
					SetDestination_Waypoint
					isArrived
					Alert
		StopAlerting
