tree("BeAlive")
	//This enemy attacks when possible,
	//otherwise it patrols a predefined path.
	fallback
		tree("Attack")
		tree("Follow")
		tree("Patrol")

tree("Attack")
	//When the enemy is visible, attack it,
	//Otherwise forget about it.
	fallback
		//Attack the enemy if visible.
		sequence
			IsPlayerInSight("Attack")
			StopAlerting
			SetDestination_Player
			Stop
			Wait(0.5)
			AimAt_Target
			Wait(0.1)
			Attack
		sequence
			StopAttacking
			Fail

tree("Follow")
	fallback
		while
			IsPlayerInSight("Follow")
			sequence
				StopAlerting
				SetDestination_Player
				Wait(0.1)
				StartNvAgent
		sequence
			StopFollowing
			Fail
		
tree("Patrol")
	//While no enemy is spotted,
	//follow the assigned waypoints
	Fallback
		sequence
			SetDestination_Waypoint
			isArrived
			Alert
		StopAlerting
