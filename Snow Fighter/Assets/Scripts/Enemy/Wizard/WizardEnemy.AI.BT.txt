tree("BeAlive")
	//This enemy attacks when possible,
	//otherwise it patrols a predefined path.
	fallback
		tree("Attack")
		tree("Follow")
		tree("Patrol")

tree("Attack")
	//When the enemy is visible, attack it,
	//Otherwise forget about it.
	fallback

		//Attack the enemy if visible.
		while
			canAttack
			sequence
				SetDestination_Player
				Stop
				Wait(0.5)
				AimAt_Target
				Wait(0.1)
				Attack
		sequence
			StopAttacking
			Fail

tree("Follow")
	fallback
		while
			canFollow
			sequence
				SetDestination_Player
				Wait(0.1)
				StartNvAgent
		sequence
			StopFollowing
			Fail
		
tree("Patrol")
	//While no enemy is spotted,
	//follow the assigned waypoints
	fallback
		while
			not canAttack
			repeat
				sequence
					SetDestination_Waypoint
					isArrived
					Alert
		StopAlerting
