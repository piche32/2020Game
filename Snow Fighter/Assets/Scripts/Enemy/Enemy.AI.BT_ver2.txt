tree("Root")
	//Root behaviour for all enemy: Be alive or die.
	parallel
		repeat mute tree("BeAlive")
		repeat mute tree("Die")

tree("BeAlive")
	//This enemy attacks when possible,
	//otherwise it patrols a predefined path.
	fallback
		tree("Attack")
		tree("Patrol")

tree("Die")
	//Die if no more HP.
	Sequence
		CheckHP
		MakeRagdoll
		repeat(15)
			sequence
				Wait(0.1)
				Blink
		Die
